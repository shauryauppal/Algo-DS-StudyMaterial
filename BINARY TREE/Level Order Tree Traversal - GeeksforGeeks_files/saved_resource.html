                                                                                                                                //level order traversing
                                                                                                                                #include<stdio.h>
                                                                                                                                #include<stdlib.h>

                                                                                                                                struct bNode
                                                                                                                                {
                                                                                                                                int data;
                                                                                                                                struct bNode *left;
                                                                                                                                struct bNode *right;
                                                                                                                                };

                                                                                                                                struct qNode
                                                                                                                                {
                                                                                                                                struct bNode* ptr;
                                                                                                                                struct qNode* next;
                                                                                                                                };

                                                                                                                                struct qNode *rear=NULL;

                                                                                                                                struct bNode* createNode(int d)
                                                                                                                                {
                                                                                                                                struct bNode* temp=(struct bNode*)malloc(sizeof(struct bNode));
                                                                                                                                if(temp==NULL)
                                                                                                                                return NULL;

                                                                                                                                temp->data=d;
                                                                                                                                temp->left=NULL;
                                                                                                                                temp->right=NULL;
                                                                                                                                return temp;
                                                                                                                                }

                                                                                                                                void enqueue(struct bNode *node, struct qNode **head)
                                                                                                                                {
                                                                                                                                if((*head)==NULL)
                                                                                                                                {
                                                                                                                                //empty queue
                                                                                                                                (*head)=(struct qNode*)malloc(sizeof(struct qNode));

                                                                                                                                (*head)->next=NULL;
                                                                                                                                (*head)->ptr=node;
                                                                                                                                rear=(*head);
                                                                                                                                return;
                                                                                                                                }

                                                                                                                                struct qNode* temp=(struct qNode*)malloc(sizeof(struct qNode));
                                                                                                                                temp->ptr=node;
                                                                                                                                temp->next=NULL;
                                                                                                                                rear->next=temp;
                                                                                                                                rear=rear->next;
                                                                                                                                return;
                                                                                                                                }

                                                                                                                                //it will deque the tree node so return type is struct bNode*
                                                                                                                                struct bNode* deque(struct qNode **head)
                                                                                                                                {

                                                                                                                                if((*head)==NULL)
                                                                                                                                {
                                                                                                                                printf("Empty queue\n");
                                                                                                                                return NULL;
                                                                                                                                }

                                                                                                                                struct bNode *b=(*head)->ptr;
                                                                                                                                //b pointing to the 1st node of queue (tree node part only)
                                                                                                                                struct qNode *q=(*head);
                                                                                                                                // q points to 1st node (tree node part + next part)
                                                                                                                                (*head)=(*head)->next;
                                                                                                                                free(q);
                                                                                                                                return b;
                                                                                                                                }

                                                                                                                                void level_order(struct bNode *root,struct qNode **q)
                                                                                                                                {
                                                                                                                                if((root)==NULL)
                                                                                                                                {
                                                                                                                                printf("NO node found\n");
                                                                                                                                return;
                                                                                                                                }

                                                                                                                                while(root)
                                                                                                                                {
                                                                                                                                printf("%d",root->data);
                                                                                                                                if(root->left)
                                                                                                                                enqueue(root->left,q);
                                                                                                                                if(root->right)
                                                                                                                                enqueue(root->right,q);

                                                                                                                                root=deque(q);

                                                                                                                                }

                                                                                                                                }

                                                                                                                                void levelorderprint(struct bNode *root)
                                                                                                                                {
                                                                                                                                struct qNode *q=NULL;
                                                                                                                                level_order(root,&q);

                                                                                                                                }
