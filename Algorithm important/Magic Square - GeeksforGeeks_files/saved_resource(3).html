                if( col == ( n - 1 ) ) col = 0;

{
   a[i--][j++]=num;
  
  if(i==-1&&j==n)      {i=i+2;    j=j-1;}
  
  else if(i==-1)       i=n-1;
  
  else if(j==n)         j=0;
  
  else if(a[i][j]!=0)  {i=i+2;   j=j-1;}
