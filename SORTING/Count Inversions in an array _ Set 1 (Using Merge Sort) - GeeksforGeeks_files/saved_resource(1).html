                                                                      dart phantom Shubham Gupta • 7 months ago

                                                                      Because the invcnt for remaining elements is already counted. Consider this example.

                                                                      L : 4 8 9 ------- R : 3 5 7

                                                                      Step 1 : 4 > 3, So count is 3 for all 4, 8 & 9

                                                                      L : 4 8 9 ------- R : 5 7

                                                                      Step 2 : 4 < 5, count is still 3.

                                                                      L : 8 9 ------- R : 5 7

                                                                      Step 3: 8 > 5, count is 3(from before) + 2(for 8, 9) = 5

                                                                      L : 8 9 ------- R : 7

                                                                      Step 4: 8 > 7, count is 5(from before) + 2(for 8, 9) = 7

                                                                      L : 8 9 ------- R : empty

                                                                      Step 5: copy remaining elements and thus resulted count is 7, since it is clear that all the inversions have been counted.

                                                                                                                                                                    A simple code than above, using inversions global variable

                                                                                                                                                                    code is below

                                                                                                                                                                    #include <cstdlib>
                                                                                                                                                                    #include <iostream>

                                                                                                                                                                    using namespace std;

                                                                                                                                                                    int b[5];
                                                                                                                                                                    int inversions=0;
                                                                                                                                                                    void merge(int *a,int left,int mid,int right)
                                                                                                                                                                    {
                                                                                                                                                                    int i,j,k;
                                                                                                                                                                    i=left;
                                                                                                                                                                    k=left;
                                                                                                                                                                    j=mid+1;
                                                                                                                                                                    while((i<=mid)&&(j<=right))
                                                                                                                                                                    {
                                                                                                                                                                    if(a[i]<=a[j])
                                                                                                                                                                    {
                                                                                                                                                                    b[k]=a[i];
                                                                                                                                                                    i++;
                                                                                                                                                                    }
                                                                                                                                                                    else
                                                                                                                                                                    {
                                                                                                                                                                    b[k]=a[j];
                                                                                                                                                                    inversions=inversions + (mid+1 - i);
                                                                                                                                                                    j++;
